CREATE TABLE catador (
    dni CHARACTER(9),
    nombre_catador VARCHAR(20),
    PRIMARY KEY (dni)
) ENGINE = INNODB;

CREATE TABLE vino (
    cod_vino INT UNSIGNED AUTO_INCREMENT,
    tipouva VARCHAR(50),
    PRIMARY KEY (cod_vino)
) ENGINE = INNODB;

CREATE TABLE califica (
    cod_califica INT UNSIGNED AUTO_INCREMENT,
    catador CHARACTER(9),
    vino INT UNSIGNED,
    puntuacion INT(3) UNSIGNED,
    PRIMARY KEY (cod_califica),
    INDEX (catador),
    FOREIGN KEY (catador) REFERENCES catador (dni) ON UPDATE CASCADE ON DELETE RESTRICT,
    INDEX (vino),
    FOREIGN KEY (vino) REFERENCES vino (cod_vino) ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE = INNODB;
